Future<List<DistrictItem>> districtList

Implementation

Future<List<DistrictItem>> get districtList {
  return platform(
    android: (pool) => _androidModel
        .getDistrict()
        .asStream()
        .asyncExpand((it) => Stream.fromIterable(it))
        .map((it) => DistrictItem.android(it))
        .toList(),
    ios: (pool) => _iosModel
        .get_districts()
        .asStream()
        .asyncExpand((it) => Stream.fromIterable(it))
        .map((it) => DistrictItem.ios(it))
        .toList(),
  );
}
