Future<List<DrivePath>> drivePathList

Implementation

Future<List<DrivePath>> get drivePathList {
  return platform(
    android: (pool) async {
      return (await _androidModel.getPaths())
          .map((it) => DrivePath.android(it))
          .toList();
    },
    ios: (pool) async {
      return (await _iosModel.get_paths())
          .map((it) => DrivePath.ios(it))
          .toList();
    },
  );
}
